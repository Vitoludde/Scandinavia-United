sweden_decision_category = {
    send_in_the_reserves = {
        allowed = {
            tag = SWE
        }

        visible = {
            always = yes
        }

        available = {
            SWE = {
                any_country = {
                    is_major = yes
                    has_war_with = SWE
                }
            }
        }

        fire_only_once = yes
        days_remove = 365

        complete_effect = {
            add_timed_idea = {
                idea = reserves_mobilized
                days = 365
            }
        }
    }
}
